local isMenuOpen = false
local isMinimized = false

-- ============================================
-- DRAG FUNCTION - Hỗ trợ Mouse và Touch
-- ============================================
local function makeDraggable(draggableObject, dragFrame)
	local dragging = false
	local dragStart, startPos
	local dragDistance = 0

	dragFrame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 
			or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = draggableObject.Position
			dragDistance = 0
			
			-- Hủy tween nếu đang chạy
			TweenService:Cancel(draggableObject)
		end
	end)

	dragFrame.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 
			or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging then
			local delta = input.Position - dragStart
			dragDistance = math.abs(delta.X) + math.abs(delta.Y)
			
			draggableObject.Position = UDim2.new(
				startPos.X.Scale, 
				startPos.X.Offset + delta.X, 
				startPos.Y.Scale, 
				startPos.Y.Offset + delta.Y
			)
		end
	end)

	return function()
		return dragDistance
	end
end

-- Toggle Button: Click để mở menu, Kéo để di chuyển
local getToggleDragDistance = makeDraggable(toggleButton, toggleButton)

-- Click để toggle menu (chỉ khi không kéo)
toggleButton.MouseButton1Click:Connect(function()
	-- Nếu kéo quá xa thì không toggle
	if getToggleDragDistance() > 10 then return end
	
	isMenuOpen = not isMenuOpen
	mainDashboard.Visible = isMenuOpen
	glowFrame.Visible = isMenuOpen

	if isMenuOpen then
		isMinimized = false
		sidebar.Visible = true
		mainPanel.Visible = true
		glowFrame.Visible = true
		
		mainDashboard.Size = UDim2.new(0, 0, 0, 0)
		mainDashboard.Position = UDim2.new(0.5, 0, 0.5, 0)
		local tween = TweenService:Create(mainDashboard, TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
			Size = UDim2.new(0, 600, 0, 400),
			Position = UDim2.new(0.5, -300, 0.5, -200)
		})
		tween:Play()
	end
end)

-- Function to switch content
local function switchContent(tabName)
	-- Hide all content
	defaultContent.Visible = false
	discordContent.Visible = false
	
	-- Show content based on tab
	if tabName == "Discord" then
		discordContent.Visible = true
	elseif tabName == "BloxFruits" then
		-- Placeholder for BloxFruits content
		defaultContent.Visible = true
		contentLabel.Text = "Tab BloxFruits\n(Đang phát triển...)"
else
		-- Default content for other tabs
		defaultContent.Visible = true
		contentLabel.Text = "Bạn đã chọn: " .. tabName
	end
end

for _, btnData in ipairs(sidebarButtons) do
	local btn = sidebar:FindFirstChild(btnData.Name)
	if btn then
		btn.MouseButton1Click:Connect(function()
			-- Switch content
			switchContent(btnData.Name)
			
			-- Highlight selected button
			for _, otherBtn in ipairs(sidebar:GetChildren()) do
				if otherBtn:IsA("TextButton") then
					otherBtn.BackgroundTransparency = 0.3
				end
			end
			btn.BackgroundTransparency = 0.8
		end)
	end
end
