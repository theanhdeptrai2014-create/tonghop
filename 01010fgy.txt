-- ============================================
-- DRAG FUNCTION - Hỗ trợ Mouse và Touch
-- ============================================
local function makeDraggable(draggableObject, dragFrame)
	local dragging = false
	local dragStart, startPos
	local dragDistance = 0

	dragFrame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = draggableObject.Position
			dragDistance = 0
			TweenService:Cancel(draggableObject)
		end
	end)

	dragFrame.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if dragging then
			local delta = input.Position - dragStart
			dragDistance = math.abs(delta.X) + math.abs(delta.Y)
			draggableObject.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
		end
	end)

	return function()
		return dragDistance
	end
end

	glowFrame.Visible = isMenuOpen

	if isMenuOpen then
		isMinimized = false
		sidebarContainer.Visible = true
		mainPanel.Visible = true
		glowFrame.Visible = true
		mainDashboard.Size = UDim2.new(0, 0, 0, 0)
		mainDashboard.Position = UDim2.new(0.5, 0, 0.5, 0)
		TweenService:Create(mainDashboard, TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
			Size = UDim2.new(0, 600, 0, 400),
			Position = UDim2.new(0.5, -300, 0.5, -200)
		}):Play()
	end
end)

-- Function to switch content
local function switchContent(tabName)
	defaultContent.Visible = false
	discordContent.Visible = false
	
	if tabName == "Discord" then
		discordContent.Visible = true
	elseif tabName == "BloxFruits" then
		defaultContent.Visible = true
		contentLabel.Text = "Tab BloxFruits\n(Đang phát triển...)"
	else
		defaultContent.Visible = true
		contentLabel.Text = "Bạn đã chọn: " .. tabName
	end
end

	if btn then
		btn.MouseButton1Click:Connect(function()
			switchContent(btnData.Name)
			
			for _, otherBtn in ipairs(sidebar:GetChildren()) do
				if otherBtn:IsA("TextButton") then
					TweenService:Create(otherBtn, TweenInfo.new(0.2), {
						BackgroundColor3 = COLORS.BgCharcoal,

					}):Play()
					local stroke = otherBtn:FindFirstChild("UIStroke")
					if stroke then stroke.Color = COLORS.MainDark end

			end
			
			TweenService:Create(btn, TweenInfo.new(0.2), {
				BackgroundColor3 = COLORS.MainDark, -- Highlight tím đậm khi được chọn
				TextColor3 = COLORS.White

	end
